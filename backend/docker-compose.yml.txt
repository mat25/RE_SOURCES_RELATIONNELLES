networks:
  internal:
    driver: bridge

services:
  database:
    image: 'mysql:8.0'
    container_name: db
    ports:
      - '3306:3306'
    networks:
      - internal
    env_file: ".env"
    volumes:
      - './data/db/:/var/lib/mysql'

  phpmyadmin:
    image: phpmyadmin/phpmyadmin
    container_name: phpmyadmin
    ports:
      - '8082:80'
    networks:
      - internal
    environment:
      - PMA_HOST=db
      - PMA_PORT=3306
    depends_on:
      - database

  api:
    build: .
    container_name: mon-api
    ports:
      - "8080:8080"
    depends_on:
      - database
    networks:
      - internal
    environment:
      SPRING_DATASOURCE_URL: jdbc:mysql://database:3306/mondb?serverTimezone=UTC
      SPRING_DATASOURCE_USERNAME: user
      SPRING_DATASOURCE_PASSWORD: password
